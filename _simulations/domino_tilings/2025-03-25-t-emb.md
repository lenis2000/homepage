// The WebAssembly module from 2025-03-25-t-emb-json.js will define "Module".
// Wait for it to initialize, so we can call doTembJSON.
Module.onRuntimeInitialized = async function() {
  // Wrap the exported WASM functions
  const doTembJSON = Module.cwrap('doTembJSON', 'number', ['number'], {async: true});
  const freeString = Module.cwrap('freeString', null, ['number']);

  // We'll draw in #t-emb-svg
  const svg = d3.select("#t-emb-svg");
  const progressEl = document.getElementById("progress");

  // Helper: call doTembJSON(n), parse the JSON
  async function getTembedding(n) {
    const ptr = await doTembJSON(n);
    const jsonStr = Module.UTF8ToString(ptr);
    freeString(ptr);
    return JSON.parse(jsonStr);
    // structure:
    // {
    //   "T": [{k, j, re, im}, ...],
    //   "O": [{k, j, re, im}, ...],
    //   "B": [{re, im}, ...]    // boundary points
    // }
  }

  // Build edges in the Aztec diamond by connecting (k,j) to neighbors (k±1,j), (k,j±1)
  // restricted to those with |k|+|j| <= n.
  function buildEdges(Tvertices) {
    // Tvertices is array of {k, j, re, im}
    // 1) Create a Map from "k,j" => array index
    const indexMap = new Map();
    Tvertices.forEach((v, idx) => {
      indexMap.set(`${v.k},${v.j}`, idx);
    });

    // 2) For each vertex, try (±1,0), (0,±1)
    const edges = [];
    const neighborSteps = [
      { dk: 1, dj: 0 },
      { dk: -1, dj: 0 },
      { dk: 0, dj: 1 },
      { dk: 0, dj: -1 },
    ];

    Tvertices.forEach((v, idx) => {
      neighborSteps.forEach(step => {
        const nk = v.k + step.dk;
        const nj = v.j + step.dj;
        const key = `${nk},${nj}`;
        if (indexMap.has(key)) {
          const nbrIdx = indexMap.get(key);
          // to avoid duplicating edges, only keep (idx < nbrIdx)
          if (nbrIdx > idx) {
            edges.push([idx, nbrIdx]);
          }
        }
      });
    });
    return edges;
  }

  // Actually draw the T-embedded Aztec diamond
  async function updateVisualization(n) {
    progressEl.textContent = `Loading T-embedding for n=${n}...`;
    svg.selectAll("*").remove(); // clear old

    let data;
    try {
      data = await getTembedding(n);
    } catch (e) {
      console.error("Error calling doTembJSON:", e);
      progressEl.textContent = "Error fetching data from WASM.";
      return;
    }

    // data has "T", "O", and "B" arrays
    const Tvertices = data.T;  // array of {k, j, re, im}
    const boundary = data.B;   // array of {re, im}

    // Build edges among T-vertices (interior edges)
    const edges = buildEdges(Tvertices);

    // Determine bounding box from T-vertices *and* boundary
    let minX = Infinity, maxX = -Infinity;
    let minY = Infinity, maxY = -Infinity;

    function updateBounds(x, y) {
      if (x < minX) minX = x;
      if (x > maxX) maxX = x;
      if (y < minY) minY = y;
      if (y > maxY) maxY = y;
    }

    Tvertices.forEach(pt => {
      updateBounds(pt.re, pt.im);
    });
    boundary.forEach(pt => {
      updateBounds(pt.re, pt.im);
    });

    const pad = 0.1; // a bit of padding
    const w = maxX - minX;
    const h = maxY - minY;
    minX -= w*pad;  maxX += w*pad;
    minY -= h*pad;  maxY += h*pad;

    // Set the SVG viewBox
    svg.attr("viewBox", [minX, -maxY, (maxX - minX), (maxY - minY)].join(" "));

    // 1) Draw the interior edges among T-vertices
    svg.append("g")
      .selectAll("line.edge")
      .data(edges)
      .join("line")
      .attr("class", "edge")
      .attr("x1", d => Tvertices[d[0]].re)
      .attr("y1", d => -Tvertices[d[0]].im)
      .attr("x2", d => Tvertices[d[1]].re)
      .attr("y2", d => -Tvertices[d[1]].im);

    // 2) Draw T-vertices
    const radius = Math.max(w, h) * 0.003;
    svg.append("g")
      .selectAll("circle.vertex")
      .data(Tvertices)
      .join("circle")
      .attr("class", "vertex")
      .attr("r", radius)
      .attr("cx", d => d.re)
      .attr("cy", d => -d.im);

    // 3) Draw the boundary polygon in red
    //    We'll just connect boundary[0]->boundary[1]->...->boundary[end].
    //    The C++ code puts them in cyclic order, including corners.
    svg.append("path")
      .attr("class", "boundary")
      .attr("d", d3.line()
                 .x(d => d.re)
                 .y(d => -d.im)
                 .curve(d3.curveLinear)
                 (boundary))
      .attr("stroke", "red")
      .attr("fill", "none");

    progressEl.textContent =
      `Done: n=${n}, T-vertices=${Tvertices.length}, edges=${edges.length}, boundary size=${boundary.length}`;
  }

  // Hook up the Update button
  document.getElementById("update-btn").addEventListener("click", () => {
    const nVal = parseInt(document.getElementById("n-input").value, 10);
    if (isNaN(nVal) || nVal < 1 || nVal > 111) {
      alert("Please pick integer 1–111.");
      return;
    }
    updateVisualization(nVal);
  });

  // Initial with n=16
  updateVisualization(16);
};
</script>
